fun sort nil = nil 
 |  sort (h::t) = 
  let 
    fun insert(i,nil) = [i]
     |   insert(i,h::t) = if i>h then i::h::t else h::insert(i,t)

  in
 insert(h,sort t) 
 end;

fun time_it (action, arg) = let
        val timer = Timer.startCPUTimer ()
        val _ = action arg
        val times = Timer.checkCPUTimer timer
    in
        Time.+ (#usr times, #sys times)
    end
